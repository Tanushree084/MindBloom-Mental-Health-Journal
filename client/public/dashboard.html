Connecting Frontend to Backend

To connect your dashboard.html to the backend API, you can use JavaScript fetch calls:

// Example: Fetch dashboard stats from your API
async function loadDashboardData() {
  try {
    const response = await fetch('/api/dashboard/stats', {
      headers: {
        'Authorization': `Bearer ${localStorage.getItem('token')}`
      }
    });
    
    if (response.ok) {
      const data = await response.json();
      // Update your dashboard with the received data
      updateDashboard(data);
    } else {
      alert('Failed to load dashboard data');
    }
  } catch (error) {
    console.error('Error:', error);
  }
}

// Call this function when the dashboard loads
loadDashboardData();
